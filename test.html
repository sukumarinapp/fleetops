$('#myMapModal').on('shown.bs.modal', function(e) { var element = $(e.relatedTarget); var data = element.data("lat").split(',') initialize(new google.maps.LatLng(data[0], data[1])); }); var map; function initialize(myCenter) { var marker = new google.maps.Marker({ position: myCenter }); var mapProp = { center: myCenter, zoom: 16, mapTypeId: google.maps.MapTypeId.ROADMAP }; map = new google.maps.Map(document.getElementById("map-canvas"), mapProp); marker.setMap(map); };
